Accordion

Static content {{group.content}}

Accordion is an AngularJS-version of bootstrap's accordion.

This directive can be used to quickly generate collapsible accordion both statically (from a template) and from the dynamic model data (using the ng-repeat directive);

Directive in the module ui.bootstrap.accordion 
<div ng-controller="AccordionDemoCtrl">
  <accordion>
    <accordion-group heading="staticTitle">Static content</accordion-group>
    <accordion-group heading="{{group.title}}" ng-repeat="group in groups">{{group.content}}</accordion-group>
  </accordion>
</div>
 
function AccordionDemoCtrl($scope) {
  $scope.staticTitle = "Static Title";

  $scope.groups = [
    { title: "Dynamic Title 1", content: "Dynamic content 1" }, 
    { title: "Dynamic Title 2", content: "Dynamic content 2" }
  ];
}

Carousel

Slide {{$index}}

{{s.text}}

 Select slide: {{$index}}

Carousel creates a carousel similar to bootstrap's image carousel.

Use a <carousel> element with <slide> elements inside it. It will automatically cycle through the slides at a given rate, and a current-index variable will be kept in sync with the currently visible slide.

Directive in the module ui.bootstrap.carousel 
<div ng-controller="CarouselDemoCtrl">
  <carousel current-index="selected" interval="5000">
    <slide ng-repeat="s in slides" active="s.active">
      <img ng-src="{{s.image}}" style="margin:auto;">
      <div class="carousel-caption">
        <h4>Slide {{$index}}</h4>
        <p>{{s.text}}</p>
      </div>
    </slide>
  </carousel>
  Select slide:
  <a ng-repeat="i in slides" ng-click="selectSlide($index)">
    {{$index}}
  </a>
</div>

function CarouselDemoCtrl($scope) {
  $scope.slides = [
    {image: 'http://placekitten.com/325/200',text: 'Kitten.'},
    {image: 'http://placekitten.com/275/200',text: 'Kitty!'},
    {image: 'http://placekitten.com/375/200',text: 'Cat.'},
    {image: 'http://placekitten.com/250/200',text: 'Feline!'}
  ];
  $scope.selectSlide = function(i) {
    $scope.selected = i;
  };
}

Dropdown Toggle

  • Click me for a dropdown, yo!
    • {{choice}}

    • DropdownToggle is a simple directive which will toggle a dropdown link on click. Simply put it on the <a> tag of the toggler-element, and it will find the nearest dropdown menu and toggle it when the <a dropdown-toggle> is clicked.

    Directive in the module ui.bootstrap.dropdownToggle 
    <li class="dropdown" ng-controller="DropdownCtrl">
  <a class="dropdown-toggle">
    Click me for a dropdown, yo!
  </a>
  <ul class="dropdown-menu">
    <li ng-repeat="choice in items">
      <a>{{choice}}</a>
    </li>
  </ul>
</li>

    function DropdownCtrl($scope) {
  $scope.items = [
    "The first choice!",
    "And another choice for you.",
    "but wait! A third!"
  ];
}

    Modal

    I'm a modal!

    with some content...

    Bootstrap's modal directive.

    Directive in the module ui.bootstrap.modal 
    <div ng-controller="ModalDemoCtrl">
    <button class="btn" ng-click="open()">Open me!</button>
    <div modal="shouldBeOpen" close="close()">
        <div class="modal-header">
            <h4>I'm a modal!</h4>
        </div>
        <div class="modal-body">
            with some content...
        </div>
        <div class="modal-footer">
            <button class="btn btn-warning cancel" ng-click="close()">Cancel</button>
        </div>
    </div>
</div>
     
    var ModalDemoCtrl = function ($scope) {

  $scope.open = function () {
    $scope.shouldBeOpen = true;
  };

  $scope.close = function () {
    $scope.closeMsg = 'I was closed at: ' + new Date();
    $scope.shouldBeOpen = false;
  };
};

    Pagination

    The selected page no: {{currentPage}}

    A lightweight pagination directive that is focused on ... providing pagination!

    It will take care of visualising a pagination bar. Additionally it will make sure that the state (enabled / disabled) of the Previous / Next buttons is maintained correctly.

    Directive in the module ui.bootstrap.pagination 
    <div ng-controller="PaginationDemoCtrl">
    <pagination num-pages="noOfPages" current-page="currentPage" class="pagination-large"></pagination>
    <pagination num-pages="noOfPages" current-page="currentPage"></pagination>
    <pagination num-pages="noOfPages" current-page="currentPage" class="pagination-small"></pagination>
    <pagination num-pages="noOfPages" current-page="currentPage" class="pagination-mini"></pagination>
    <button class="btn" ng-click="setPage(3)">Set current page to: 3</button>
    The selected page no: {{currentPage}}
</div>
     
    var PaginationDemoCtrl = function ($scope) {
  $scope.noOfPages = 5;
  $scope.currentPage = 4;

  $scope.setPage = function (pageNo) {
    $scope.currentPage = pageNo;
  };
};
